# app/views.py
```python
from django.shortcuts import render

from .models import Category


def category_detail_view(request, pk):
    category = Category.objects.get(pk=pk)
    return render(request, "app/category_detail.html", {"category": category})
```

```html
<!-- app/templates/app/category_detail.html -->

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Category details</title>
</head>
<body>
    <h1>{{ category.name }}</h1>
</body>
</html>
```

# app/urls.py
```python
from django.urls import path

from .views import category_detail_view

urlpatterns = [
    path("categories/<int:pk>/", category_detail_view, name="category_detail"),
]
```
